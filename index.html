# Define a function to determine Pass or Fail
pass_fail <- function(marks_vector) {
# Initialize an empty vector to store Pass/Fail results
result_vector <- vector("character", length(marks_vector))

# Loop through each mark
for (i in 1:length(marks_vector)) {
    if (marks_vector[i] >= 50) {
    result_vector[i] <- "Pass"
    } else {
    result_vector[i] <- "Fail"
    }
}

return(result_vector)
}

# Example usage:
marks <- c(65, 42, 78, 30, 55)
pass_fail_result <- pass_fail(marks)
print(pass_fail_result)


#2
pass_fail_anonymous <- function(marks_vector) {
return(ifelse(marks_vector >= 50, "Pass", "Fail"))
}

# Example usage:
marks <- c(65, 42, 78, 30, 55)
pass_fail_result_anonymous <- pass_fail_anonymous(marks)
print(pass_fail_result_anonymous)
